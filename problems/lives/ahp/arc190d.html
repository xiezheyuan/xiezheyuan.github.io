[Medium-Hard] AT_arc190_d [ARC190D] Matrix Pow Sum

简要题意

给定一个 n×nn\times nn×n 的方阵 AAA，若 Aij=0A_{ij}=0Aij​=0，表示 AijA_{ij}Aij​ 不确定。

给出整数 ppp，你需要将每个不确定的位置填上一个 [1,p)[1,p)[1,p) 的整数得到矩阵 BBB。求全体 BpB^pBp 的和。答案对 ppp 取模。

1≤n≤100,p∈P,p≤1091\leq n\leq 100,p\in\mathbb{P},p\leq 10^91≤n≤100,p∈P,p≤109。

对于 p=2p=2p=2，求解是平凡的，这里只考虑 p>2p>2p>2 的情况。

不妨将每个不确定的位置 AijA_{ij}Aij​ 看成一个自由元 xijx_{ij}xij​。先考察 AijkA_{ij}^kAijk​，则每个位置都是一个多元至多 ppp 次的多项式。

考虑每一个单元格的多项式的每一项，应当形如 Cx1k1x2k2⋯C x_1^{k_1} x_2^{k_2} \cdotsCx1k1​​x2k2​​⋯，则这一项产生的贡献也应当形如 C(∑i=1p−1ik1)(∑i=1p−1ik2)⋯C\left(\sum_{i=1}^{p-1} i^{k_1}\right)\left(\sum_{i=1}^{p-1} i^{k_2}\right)\cdotsC(∑i=1p−1​ik1​)(∑i=1p−1​ik2​)⋯。这启发我们往 Fp(k)=(∑i=1p−1ik) mod pF_p(k)=\left(\sum_{i=1}^{p-1} i^k\right)\bmod pFp​(k)=(∑i=1p−1​ik)modp 去思考。

看起来十分困难，不妨模拟矩阵乘法的过程，分析每一项是如何得到的。

对于 p>3p>3p>3，我们只有下面的两种方法：

  • xii⋅⋯⋅xii⋅Aijx_{ii}\cdot \cdots\cdot x_{ii}\cdot A_{ij}xii​⋅⋯⋅xii​⋅Aij​。

时间复杂度 O(MatrixMul(n)log⁡p)O(\mathrm{MatrixMul}(n)\log p)O(MatrixMul(n)logp)。

代码

#include <bits/stdc++.h>
using namespace std;

int n, mod;

int Add(int x, int y){ return (x + y) >= mod ? (x + y - mod) : (x + y); }
int Sub(int x, int y){ return (x - y) < 0 ? (x - y + mod) : (x - y); }
int Mul(int x, int y){ return 1ll * x * y % mod; }
int fastpow(int x, int y){
    int ret = 1;
    for(;y;y>>=1,x=Mul(x, x)){ if(y & 1) ret = Mul(ret, x); }
    return ret;
}

const int N = 105;

struct Matrix {
    int a[N][N];
    Matrix(){ memset(a, 0, sizeof(a)); }
    int* operator[](int x){ return a[x]; }
    void idt(){
        for(int i=1;i<=n;i++) a[i][i] = 1;
    }
} A, B;

Matrix operator*(Matrix A, Matrix B){
    Matrix C;
    for(int i=1;i<=n;i++){
        for(int k=1;k<=n;k++){
            for(int j=1;j<=n;j++) C[i][j] = Add(C[i][j], Mul(A[i][k], B[k][j]));
        }
    }
    return C;
}

Matrix operator^(Matrix A, int k){
    Matrix B; B.idt();
    while(k){
        if(k & 1) B = B * A;
        A = A * A, k >>= 1;
    }
    return B;
}

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> mod; int cnt = 0;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++) cin >> A[i][j], cnt += (A[i][j] == 0);
    }
    if(mod == 2){
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++) A[i][j] = 1;
        }
        B = A ^ mod;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++) cout << B[i][j] << ' ';
            cout << '\n';
        }
        return 0;
    }
    B = A ^ mod;
    for(int i=1;i<=n;i++){
        if(A[i][i]) continue;
        for(int j=1;j<=n;j++){
            B[i][j] = Add(B[i][j], A[i][j]);
            B[j][i] = Add(B[j][i], A[j][i]);
        }
    }
    if(mod == 3){
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                if(A[i][j]) continue;
                B[i][j] = Add(B[i][j], A[j][i]);
            }
        }
    }
    int tmp = cnt & 1 ? mod - 1 : 1;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++) cout << Mul(B[i][j], tmp) << ' ';
        cout << '\n';
    }
    return 0;
}
