        g[u][v] = 1, ind[v] |= (1 << (u - 1)), outd[u] |= (1 << (v - 1));
    }
    // Step 0: é¢„å¤„ç† 2^n
    pw2[0] = 1;
    for(int i=1;i<=m;i++) pw2[i] = Add(pw2[i - 1], pw2[i - 1]);
    // Step 1: é¢„å¤„ç† p(S) è¡¨ç¤ºç‚¹é›† S çš„å¯¼å‡ºå­å›¾è¾¹æ•°
    for(int i=1;i<(1<<n);i++){
        int x = lowbit(i), px = __lg(x) + 1;
        p[i] = p[i ^ x];
        for(int j=1;j<=n;j++){
            if(((i ^ x) >> (j - 1)) & 1) p[i] += g[px][j], p[i] += g[j][px];
        }
    }
    for(int S=1;S<(1<<n);S++){
        if(popcnt(S) == 1){
            f[S] = 1, h[S] = mod - 1;
            continue;
        }
        // Step 2ï¼šé€’æ¨ zak[T] è¡¨ç¤ºèµ·ç‚¹åœ¨ Tï¼Œç»ˆç‚¹åœ¨ S-T çš„è¾¹æ•°
        stack<int> stk;
        for(int T=S;T;T=(T-1)&S) stk.push(T);
        for(;!stk.empty();stk.pop()){
            int T = stk.top(), u = lowbit(T), pu = __lg(u) + 1;
            zak[T] = zak[T ^ u] - popcnt(ind[pu] & (T ^ u)) + popcnt(outd[pu] & (S ^ T));
        }
        // Step 4ï¼šé€’æ¨ h[S] è¡¨ç¤ºå®¹æ–¥ç³»æ•°
        int x = lowbit(S);
        for(int T=S^x;T;T=(T-1)&(S^x)) h[S] = Sub(h[S], Mul(h[T], f[S ^ T]));
        // Step 3: é€’æ¨ f[S] è¡¨ç¤ºç‚¹é›†ä¸º S çš„ç­”æ¡ˆ
        f[S] = Add(pw2[p[S]], h[S]);
        for(int T=S;T;T=(T-1)&S){
            if(T == S) continue;
            f[S] = Add(f[S], Mul(h[T], pw2[zak[T] + p[S ^ T]]));
        }
        h[S] = Sub(h[S], f[S]);
    }
    cout << f[(1 << n) - 1] << '\n';
    return 0;
}
